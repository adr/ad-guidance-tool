# MADR 4.0 File Format & Repository — PR 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace ADG's HTML-anchor-based file format with MADR 4.0–native markdown, ported end-to-end through the domain types, parser, renderer, repository, validator, and `models/clean` fixture. After this PR, every `adg` command writes/reads MADR-shaped files. UX improvements (renamed flags, stdout/stderr split, new commands) are deferred to PR 2.

**Architecture:** Frontmatter (YAML) holds metadata + ADG extensions (tags, custom links, comments). Body is pure MADR markdown with no HTML anchors. Comments are sourced from frontmatter and rendered into a trailing `## Comments` body section on every write — the body section is never parsed back, eliminating the legacy comment-rewrite data-loss bug class. Sections are matched by case-insensitive H2 header text. `index.yaml` is **dropped** in this PR; commands scan the model directory directly.

**Tech Stack:** Go 1.22, `gopkg.in/yaml.v3`, `github.com/spf13/cobra`, `github.com/stretchr/testify/assert`, mockery for repository mocks. Clean Architecture layout already in place.

**Reference spec:** `docs/superpowers/specs/2026-05-13-adg-madr-fork-design.md`

## Pre-flight ratifications (confirm before starting Task 1)

The spec leaves two design calls "to ratify before PR 1". This plan commits to:

1. **`index.yaml` — DROPPED.** ADR files are the only source of truth. Commands scan the directory. `adg rebuild` becomes a no-op stub for PR 1 (printed friendly message) and may be removed in PR 2. Rationale: simpler architecture, fewer moving parts, fast enough for models under ~1000 ADRs, can be added back as `--cache` opt-in if needed.

2. **`adg revise` and supersession — independent.** `revise` creates a new ADR in `proposed` state with no supersession relationship. The user runs `adg supersede --new X --old Y` separately (PR 2 work) to record the relationship.

If either choice should flip, raise it before Task 1.

```go
// Decision is the in-memory representation of an Architectural Decision Record.
//
// ID, Slug, and Title are derived (filename + H1) and not stored in frontmatter.
// All other fields are persisted to frontmatter via Frontmatter().
type Decision struct {
	// Identity (from filename)
	ID   string
	Slug string

	// Title (from H1)
	Title string

	// MADR frontmatter
	Status         string
	Date           string
	DecisionMakers []string
	Consulted      []string
	Informed       []string

	// ADG extensions
	Tags          []string
	Links         map[string][]string
	Supersedes    []string
	Comments      []Comment
	LegacyOutcome bool
}

// Comment is one entry in the ADG-extension `comments` frontmatter list.
// Date is a timestamp with time-of-day (YYYY-MM-DD HH:MM:SS) to preserve
// ordering within a single day.
type Comment struct {
	Author string `yaml:"author"`
	Date   string `yaml:"date"`
	Text   string `yaml:"text"`
}

// Frontmatter is the YAML-serializable shape persisted at the top of an ADR file.
type Frontmatter struct {
	Status         string              `yaml:"status,omitempty"`
	Date           string              `yaml:"date,omitempty"`
	DecisionMakers []string            `yaml:"decision-makers,omitempty"`
	Consulted      []string            `yaml:"consulted,omitempty"`
	Informed       []string            `yaml:"informed,omitempty"`
	Tags           []string            `yaml:"tags,omitempty"`
	Links          map[string][]string `yaml:"links,omitempty"`
	Supersedes     []string            `yaml:"supersedes,omitempty"`
	Comments       []Comment           `yaml:"comments,omitempty"`
	LegacyOutcome  bool                `yaml:"legacy-outcome,omitempty"`
}

	return Frontmatter{

		Date:           d.Date,

		Consulted:      d.Consulted,

		Comments:       d.Comments,

		Informed:       fm.Informed,

		Links:          fm.Links,
```

```go
package decision

import (
	"bytes"
	"fmt"
)

// SplitFile separates the optional YAML frontmatter (between `---` fences at the
// very top) from the markdown body. Returns frontmatter text without the fences
// (may be empty if no frontmatter), body text, or an error if the frontmatter
// is opened but never closed.
func SplitFile(content []byte) (frontmatter, body string, err error) {
	content = bytes.ReplaceAll(content, []byte("\r\n"), []byte("\n"))

	// No frontmatter case: file doesn't start with "---\n"
	if !bytes.HasPrefix(content, []byte("---\n")) {
		return "", string(content), nil
	}

	rest := content[len("---\n"):]
	closeIdx := bytes.Index(rest, []byte("\n---\n"))
	if closeIdx == -1 {
		// Tolerate file ending with "\n---" (no trailing newline)
		if bytes.HasSuffix(rest, []byte("\n---")) {
			closeIdx = len(rest) - len("\n---")
			return string(rest[:closeIdx]), "", nil
		}
		return "", "", fmt.Errorf("frontmatter opened with `---` but never closed")
	}

	fm := string(rest[:closeIdx+1]) // include trailing newline inside fence
	bodyStart := closeIdx + len("\n---\n")
	return fm, string(rest[bodyStart:]), nil
}
```

```go
import (
	// ... existing imports ...
	"regexp"
	"strings"
)

// ParsedBody is the output of ParseBody — everything we extracted from a body.
type ParsedBody struct {
	Title            string
	Sections         map[string]string // canonical key (lowercase) -> raw section text including the H2 line
	Options          []string          // bullet items under Considered Options, in order
	ChosenOption     string            // text from `Chosen option: "..."`
	OutcomeRationale string            // text after `because ` and before the trailing `.`
	CustomSections   map[string]string // unrecognized H2 header text -> raw section text
}

// Maps canonical section keys to the lowercase header text we accept.
// Matching is case-insensitive contains-style is NOT used — we want exact
// header text equality (case-insensitive) so a custom "Considered Trade-offs"
// section isn't mistaken for "Considered Options".
var canonicalSections = map[string]string{
	"context and problem statement": "context",
	"decision drivers":              "drivers",
	"considered options":            "options",
	"decision outcome":              "outcome",
	"pros and cons of the options":  "pros-cons",
	"more information":              "more",
	"comments":                      "comments",
}

var h1Re = regexp.MustCompile(`(?m)^# +(.+)$`)
var h2Re = regexp.MustCompile(`(?m)^## +(.+?)\s*$`)
var bulletRe = regexp.MustCompile(`(?m)^\s*\*\s+(.+)$`)
var chosenRe = regexp.MustCompile(`(?m)^Chosen option:\s*"([^"]*)"(?:\s*,\s*because\s+(.+?))?\.?\s*$`)

func ParseBody(body string) (*ParsedBody, error) {
	pb := &ParsedBody{
		Sections:       map[string]string{},
		CustomSections: map[string]string{},
	}

	// Title (H1)
	if m := h1Re.FindStringSubmatch(body); m != nil {
		pb.Title = strings.TrimSpace(m[1])
	}

	// Walk H2 boundaries and build section slices.
	h2Indexes := h2Re.FindAllStringSubmatchIndex(body, -1)
	for i, idx := range h2Indexes {
		start := idx[0]
		end := len(body)
		if i+1 < len(h2Indexes) {
			end = h2Indexes[i+1][0]
		}
		section := body[start:end]

		headerText := strings.TrimSpace(body[idx[2]:idx[3]])
		key, isCanonical := canonicalSections[strings.ToLower(headerText)]
		if isCanonical {
			pb.Sections[key] = section
		} else {
			pb.CustomSections[headerText] = section
		}
	}

	// Options: bullets under Considered Options.
	if opts, ok := pb.Sections["options"]; ok {
		for _, m := range bulletRe.FindAllStringSubmatch(opts, -1) {
			pb.Options = append(pb.Options, strings.TrimSpace(m[1]))
		}
	}

	// Chosen option from Decision Outcome.
	if outcome, ok := pb.Sections["outcome"]; ok {
		if m := chosenRe.FindStringSubmatch(outcome); m != nil {
			pb.ChosenOption = m[1]
			pb.OutcomeRationale = strings.TrimSpace(m[2])
		}
	}

	return pb, nil
}
```

```go
func ParseFrontmatter(text string) (Frontmatter, error) {
	if strings.TrimSpace(text) == "" {
		return Frontmatter{}, nil

	if err := yaml.Unmarshal([]byte(text), &fm); err != nil {
		return Frontmatter{}, fmt.Errorf("invalid frontmatter YAML: %w", err)

	return fm, nil
}
```

```go
// IsLegacyADG returns true if the file appears to use the pre-MADR ADG format.
// Read-side commands use this to refuse legacy files and steer users to
// `adg migrate`. Detection rules per the design spec.
func IsLegacyADG(path string, content []byte) bool {
	base := filepath.Base(path)
	if strings.HasPrefix(base, "AD") && strings.HasSuffix(base, ".md") {
		if regexp.MustCompile(`^AD\d{4}-.*\.md$`).MatchString(base) {

		}

	if legacyAnchorRe.Match(content) {
		return true
	}
	if legacyStatusRe.Match(content) {
		return true
	}
	if legacyADRIDRe.Match(content) {
		return true
	}
	return false
}
```

```go
const canonicalTemplate = `# %s

## Context and Problem Statement

{...}

## Decision Drivers

* {driver 1}

## Considered Options

* {option 1}

Chosen option: "{option title}", because {justification}.

// RenderNewBody emits the canonical minimal+Decision-Drivers template for a
// freshly-created ADR. The title is interpolated into the H1.
func RenderNewBody(title string) string {
	return fmt.Sprintf(canonicalTemplate, title)

// renderCommentsSection produces the trailing ## Comments H2 from a Comment list.
// Returns "" if the list is empty (the section is omitted entirely).
func renderCommentsSection(comments []Comment) string {
	if len(comments) == 0 {
		return ""

	var b strings.Builder
	b.WriteString("## Comments\n\n")
	for _, c := range comments {
		b.WriteString(fmt.Sprintf("* **%s — @%s:** %s\n", c.Date, c.Author, c.Text))
	}
	return b.String()
}
```

```go
import (
	// ... existing ...
	"bytes"
	"gopkg.in/yaml.v3"
)

// RenderFile assembles the on-disk bytes for an ADR: optional YAML frontmatter
// between `---` fences, then the body. Any existing `## Comments` section in
// the body is stripped, and a fresh one is rendered from d.Comments at the end.
//
// If d has no populated frontmatter fields, no frontmatter block is emitted —
// MADR's minimal template is frontmatter-free, and this respects that case.
func RenderFile(d Decision, body string) (string, error) {
	stripped := stripCommentsSection(body)
	stripped = strings.TrimRight(stripped, "\n") + "\n"

	commentsSection := renderCommentsSection(d.Comments)

	fm := d.Frontmatter()
	fmBytes, err := yaml.Marshal(fm)
	if err != nil {
		return "", fmt.Errorf("failed to marshal frontmatter: %w", err)
	}

	hasFrontmatter := len(bytes.TrimSpace(fmBytes)) > 2 // "{}\n" when empty

	var out bytes.Buffer
	if hasFrontmatter {
		out.WriteString("---\n")
		out.Write(fmBytes)
		out.WriteString("---\n\n")
	}
	out.WriteString(stripped)
	if commentsSection != "" {
		out.WriteString("\n")
		out.WriteString(commentsSection)
	}
	return out.String(), nil
}

// stripCommentsSection removes the `## Comments` H2 and its contents from a body.
// Anything after the next H2 (or EOF) is preserved.
func stripCommentsSection(body string) string {
	lines := strings.Split(body, "\n")
	var out []string
	skipping := false
	for _, line := range lines {
		if !skipping && strings.HasPrefix(strings.ToLower(strings.TrimSpace(line)), "## comments") {
			skipping = true
			continue
		}
		if skipping && strings.HasPrefix(strings.TrimSpace(line), "## ") {
			skipping = false
		}
		if !skipping {
			out = append(out, line)
		}
	}
	return strings.Join(out, "\n")
}
```

```go
type FileDecisionRepository struct{}

func NewFileDecisionRepository() *FileDecisionRepository {

}

func (r *FileDecisionRepository) Create(modelPath, subFolderPath string, d *domain.Decision) (*domain.Decision, error) {
	id, err := r.generateNextID(modelPath)
	if err != nil {
		return nil, err
	}
	d.ID = id
	d.Slug = slugify(d.Title)

	body := domain.RenderNewBody(d.Title)
	out, err := domain.RenderFile(*d, body)
	if err != nil {
		return nil, err
	}

	filename := fmt.Sprintf("%s-%s.md", d.ID, d.Slug)
	fullPath := filepath.Join(modelPath, subFolderPath, filename)
	if err := os.MkdirAll(filepath.Dir(fullPath), 0o755); err != nil {
		return nil, fmt.Errorf("failed to create dir: %w", err)
	}
	if err := os.WriteFile(fullPath, []byte(out), 0o644); err != nil {
		return nil, fmt.Errorf("failed to write file: %w", err)
	}
	return d, nil
}

func slugify(title string) string {
	s := strings.ToLower(title)
	s = strings.ReplaceAll(s, " ", "-")
	// retain only [a-z0-9-]
	var b strings.Builder
	for _, r := range s {
		if (r >= 'a' && r <= 'z') || (r >= '0' && r <= '9') || r == '-' {
			b.WriteRune(r)
		}
	}
	return b.String()
}

func (r *FileDecisionRepository) generateNextID(modelPath string) (string, error) {
	maxID := 0
	err := filepath.WalkDir(modelPath, func(path string, d fs.DirEntry, err error) error {
		if err != nil || d.IsDir() {
			return err
		}
		name := d.Name()
		if len(name) < 5 || !strings.HasSuffix(name, ".md") {
			return nil
		}
		// Accept NNNN-... format only (legacy AD-prefixed files ignored)
		if id, err := strconv.Atoi(name[:4]); err == nil && len(name) > 4 && name[4] == '-' {
			if id > maxID {
				maxID = id
			}
		}
		return nil
	})
	if err != nil && !os.IsNotExist(err) {
		return "", fmt.Errorf("scan failed: %w", err)
	}
	return fmt.Sprintf("%04d", maxID+1), nil
}

func (r *FileDecisionRepository) FindDecisionFile(modelPath, id string) (string, error) {
	var found string
	prefix := id + "-"
	err := filepath.WalkDir(modelPath, func(path string, d fs.DirEntry, err error) error {
		if err != nil || d.IsDir() {
			return err
		}
		if strings.HasPrefix(d.Name(), prefix) && strings.HasSuffix(d.Name(), ".md") {
			found = path
			return io.EOF
		}
		return nil
	})
	if err != nil && err != io.EOF {
		return "", err
	}
	if found == "" {
		return "", fmt.Errorf("decision %s not found", id)
	}
	return found, nil
}
```

```go
	all, err := r.LoadAll(modelPath)
	if err != nil {
		return nil, err
	}
	slug := slugify(title)
	var exact *domain.Decision
	var partial []*domain.Decision
	for i := range all {
		d := &all[i]
		switch {
		case d.Slug == slug:
			if exact != nil {
				return nil, fmt.Errorf("multiple decisions match title exactly; use id")
			}
			exact = d
		case strings.Contains(d.Slug, slug):
			partial = append(partial, d)
		}
	}
	switch {
	case exact != nil:
		return exact, nil
	case len(partial) == 1:
		return partial[0], nil
	case len(partial) > 1:
		return nil, fmt.Errorf("multiple titles matched %q; be more specific or use id", title)
	default:
		return nil, fmt.Errorf("no decision title matched %q", title)
	}
}

func (r *FileDecisionRepository) LoadAll(modelPath string) ([]domain.Decision, error) {

		}

			return nil

		}
		d, err := r.loadFile(path)
		if err != nil {
			return err
		}
		d.ID = id
		decisions = append(decisions, *d)
		return nil
	})
	if err != nil {

	}

		return nil, fmt.Errorf("file %s appears to use legacy ADG format; run 'adg migrate' to convert", path)

		return nil, err

		return nil, err
	}

	d.ID = id
	d.Slug = slug
	d.Title = parsed.Title
	return &d, nil
```

```go
		return fmt.Errorf("failed to compute relative path: %w", err)
	}
	dstFile := filepath.Join(dstPath, rel)
	if err := os.MkdirAll(filepath.Dir(dstFile), 0o755); err != nil {
		return err
	}
	content, err := os.ReadFile(srcFile)
	if err != nil {
		return err
	}
	return os.WriteFile(dstFile, content, 0o644)
}
```

```go
type DecisionService interface {
	AddNew(modelPath, title string) (*Decision, error)

func (s *DecisionServiceImplementation) Link(modelPath string, source, target *Decision, tag, reverseTag string) error {
	if tag == "supersedes" || tag == "superseded-by" {
		return fmt.Errorf("use 'adg supersede' to record supersession, not 'adg link'")
	}
	if source.Links == nil {
		source.Links = map[string][]string{}
	}
	if !slices.Contains(source.Links[tag], target.ID) {
		source.Links[tag] = append(source.Links[tag], target.ID)
	}
	source.Date = time.Now().Format("2006-01-02")

	if reverseTag != "" {
		if target.Links == nil {
			target.Links = map[string][]string{}
		}
		if !slices.Contains(target.Links[reverseTag], source.ID) {
			target.Links[reverseTag] = append(target.Links[reverseTag], source.ID)
		}
		target.Date = time.Now().Format("2006-01-02")
	}

	bodySrc, err := s.repo.LoadBody(modelPath, source.ID)
	if err != nil {
		return err
	}
	if err := s.repo.Save(modelPath, source, bodySrc); err != nil {
		return err
	}
	if reverseTag != "" {
		bodyTgt, err := s.repo.LoadBody(modelPath, target.ID)
		if err != nil {
			return err
		}
		return s.repo.Save(modelPath, target, bodyTgt)
	}
	return nil
}

func (s *DecisionServiceImplementation) Decide(modelPath string, d *Decision, option, rationale string) error {
	body, err := s.repo.LoadBody(modelPath, d.ID)
	if err != nil {
		return err
	}
	parsed, err := ParseBody(body)
	if err != nil {
		return err
	}

	// Resolve option: numeric -> index into parsed.Options; text -> case-insensitive match.
	var chosen string
	if n, err := strconv.Atoi(option); err == nil {
		if n < 1 || n > len(parsed.Options) {
			return fmt.Errorf("option %d out of range (1..%d); use 'adg edit' to add the option first", n, len(parsed.Options))
		}
		chosen = parsed.Options[n-1]
	} else {
		target := strings.ToLower(strings.TrimSpace(option))
		for _, o := range parsed.Options {
			if strings.ToLower(o) == target {
				chosen = o
				break
			}
		}
		if chosen == "" {
			return fmt.Errorf("option %q is not in Considered Options; use 'adg edit' to add it first", option)
		}
	}

	// Replace Decision Outcome section's body with the Chosen-option line.
	newBody, err := replaceOutcomeSection(body, chosen, rationale)
	if err != nil {
		return err
	}

	d.Status = "accepted"
	d.Date = time.Now().Format("2006-01-02")
	d.LegacyOutcome = false
	return s.repo.Save(modelPath, d, newBody)
}

func replaceOutcomeSection(body, option, rationale string) (string, error) {
	var newOutcome string
	if rationale != "" {
		newOutcome = fmt.Sprintf("## Decision Outcome\n\nChosen option: %q, because %s.\n", option, rationale)
	} else {
		newOutcome = fmt.Sprintf("## Decision Outcome\n\nChosen option: %q.\n", option)
	}
	return replaceH2Section(body, "decision outcome", newOutcome), nil
}

// replaceH2Section finds the H2 with the given lowercase header text and replaces
// it (and its contents up to the next H2 or EOF) with the supplied newSection
// (which must begin with `## ...`). If no matching H2 is found, newSection is
// appended after all existing H2s (canonical MADR ordering is the renderer's job).
func replaceH2Section(body, lowercaseHeader, newSection string) string {
	lines := strings.Split(body, "\n")
	var out []string
	skipping := false
	found := false
	for _, line := range lines {
		trimmed := strings.ToLower(strings.TrimSpace(line))
		if !skipping && strings.HasPrefix(trimmed, "## "+lowercaseHeader) {
			skipping = true
			found = true
			out = append(out, strings.TrimRight(newSection, "\n"))
			continue
		}
		if skipping && strings.HasPrefix(strings.TrimSpace(line), "## ") {
			skipping = false
		}
		if !skipping {
			out = append(out, line)
		}
	}
	if !found {
		// Append a new section.
		out = append(out, "", strings.TrimRight(newSection, "\n"))
	}
	return strings.Join(out, "\n")
}

// EditAppend appends content to the named MADR section. If the section doesn't exist,
// it's created in canonical MADR order via the renderer's section-ordering pass.
func (s *DecisionServiceImplementation) EditAppend(modelPath string, d *Decision, section, content string) error {
	body, err := s.repo.LoadBody(modelPath, d.ID)
	if err != nil {
		return err
	}
	header, ok := sectionHeader(section)
	if !ok {
		return fmt.Errorf("unknown section: %s", section)
	}
	body = appendToH2Section(body, header, content)
	d.Date = time.Now().Format("2006-01-02")
	return s.repo.Save(modelPath, d, body)
}

func sectionHeader(section string) (string, bool) {
	switch section {
	case "context":
		return "## Context and Problem Statement", true
	case "drivers":
		return "## Decision Drivers", true
	case "options":
		return "## Considered Options", true
	case "more":
		return "## More Information", true
	}
	return "", false
}

func appendToH2Section(body, header, content string) string {
	if strings.Contains(body, header) {
		// Insert just before the next H2 (or EOF).
		lines := strings.Split(body, "\n")
		startIdx := -1
		for i, line := range lines {
			if strings.HasPrefix(line, header) {
				startIdx = i
				break
			}
		}
		insertAt := len(lines)
		for i := startIdx + 1; i < len(lines); i++ {
			if strings.HasPrefix(strings.TrimSpace(lines[i]), "## ") {
				insertAt = i
				break
			}
		}
		newLines := append([]string{}, lines[:insertAt]...)
		newLines = append(newLines, content)
		newLines = append(newLines, lines[insertAt:]...)
		return strings.Join(newLines, "\n")
	}
	// Section absent — append at end.
	return strings.TrimRight(body, "\n") + "\n\n" + header + "\n\n" + content + "\n"
}

func (s *DecisionServiceImplementation) Revise(modelPath string, original *Decision) (*Decision, error) {
	body, err := s.repo.LoadBody(modelPath, original.ID)
	if err != nil {
		return nil, err
	}
	// Clear Decision Outcome and any H3 Consequences.
	body = replaceH2Section(body, "decision outcome", "## Decision Outcome\n\n{...}\n")

	today := time.Now().Format("2006-01-02")
	revised := &Decision{
		Title:          original.Title + " (Revised)",
		Status:         "proposed",
		Date:           today,
		DecisionMakers: original.DecisionMakers,
		Consulted:      original.Consulted,
		Informed:       original.Informed,
		Tags:           original.Tags,
		Links:          nil,
		Supersedes:     nil,
		Comments:       nil,
	}
	created, err := s.repo.Create(modelPath, "", revised)
	if err != nil {
		return nil, err
	}
	// Replace the freshly-created body with the (cleared) source body, retitled.
	body = replaceH1Title(body, created.Title)
	if err := s.repo.Save(modelPath, created, body); err != nil {
		return nil, err
	}
	return created, nil
}

func replaceH1Title(body, newTitle string) string {
	return regexp.MustCompile(`(?m)^# +.+$`).ReplaceAllString(body, "# "+newTitle)
}

func (s *DecisionServiceImplementation) Copy(srcPath, dstPath, decisionID string) error {
	return s.repo.Copy(srcPath, dstPath, decisionID)
}

// FilterDecisions is unchanged in shape from the legacy implementation; it operates
// purely on already-loaded Decision slices.
func (s *DecisionServiceImplementation) FilterDecisions(decisions []Decision, filters map[string][]string) ([]Decision, error) {
	var results []Decision
	idSet := map[string]bool{}
	if idFilters, ok := filters["id"]; ok {
		expanded, err := expandIDFilters(idFilters)
		if err != nil {
			return nil, err
		}
		for _, id := range expanded {
			idSet[id] = true
		}
	}
	var titleRegex *regexp.Regexp
	if titles, ok := filters["title"]; ok && len(titles) > 0 {
		var err error
		titleRegex, err = regexp.Compile(titles[0])
		if err != nil {
			return nil, fmt.Errorf("invalid title regex: %w", err)
		}
	}
	for _, d := range decisions {
		if matchesID(d, idSet) || matchesTitle(d, titleRegex) || matchesTag(d, filters["tag"]) || matchesStatus(d, filters["status"]) {
			results = append(results, d)
		}
	}
	return results, nil
}

// Helpers
func containsLetter(s string) bool {
	matched, err := regexp.MatchString(`[a-zA-Z]`, s)
	return err == nil && matched
}

func matchesID(d Decision, idSet map[string]bool) bool {
	return len(idSet) > 0 && idSet[d.ID]
}

func matchesTitle(d Decision, titleRegex *regexp.Regexp) bool {
	return titleRegex != nil && titleRegex.MatchString(d.Title)
}

func matchesTag(d Decision, tags []string) bool {
	for _, ft := range tags {
		if slices.Contains(d.Tags, ft) {
			return true
		}
	}
	return false
}

func matchesStatus(d Decision, statuses []string) bool {
	for _, s := range statuses {
		if d.Status == s {
			return true
		}
	}
	return false
}

func expandIDFilters(ids []string) ([]string, error) {
	var result []string
	for _, raw := range ids {
		for _, id := range strings.Split(raw, ",") {
			id = strings.TrimSpace(id)
			if strings.Contains(id, "-") {
				expanded, err := expandRange(id)
				if err != nil {
					return nil, err
				}
				result = append(result, expanded...)
			} else {
				result = append(result, id)
			}
		}
	}
	return result, nil
}

func expandRange(rng string) ([]string, error) {
	parts := strings.Split(rng, "-")
	if len(parts) != 2 {
		return nil, fmt.Errorf("invalid ID range: %s", rng)
	}
	start, err1 := strconv.Atoi(parts[0])
	end, err2 := strconv.Atoi(parts[1])
	if err1 != nil || err2 != nil || start > end {
		return nil, fmt.Errorf("invalid ID range: %s", rng)
	}
	var out []string
	for i := start; i <= end; i++ {
		out = append(out, fmt.Sprintf("%04d", i))
	}
	return out, nil
}
```

```go
// Add (or replace existing) inside ModelServiceImplementation:

func (s *ModelServiceImplementation) Validate(modelPath string) ([]ValidationIssue, error) {
	decisions, err := s.decisionRepo.LoadAll(modelPath)
	if err != nil {
		return nil, err
	}

	byID := map[string]decisiondomain.Decision{}
	for _, d := range decisions {
		byID[d.ID] = d
	}

	var issues []ValidationIssue
	for _, d := range decisions {
		issues = append(issues, validateDecision(d, byID, modelPath)...)
	}
	return issues, nil
}

type ValidationIssue struct {
	ID      string
	Message string
}

var statusRe = regexp.MustCompile(`^(proposed|rejected|accepted|deprecated|superseded by ADR-[0-9]{4})$`)
var supersededRe = regexp.MustCompile(`^superseded by ADR-([0-9]{4})$`)

func validateDecision(d decisiondomain.Decision, byID map[string]decisiondomain.Decision, modelPath string) []ValidationIssue {
	var issues []ValidationIssue
	add := func(msg string) { issues = append(issues, ValidationIssue{ID: d.ID, Message: msg}) }

	// 1. Filename pattern (already validated by repo.LoadAll, but guard anyway).
	if d.ID == "" || !regexp.MustCompile(`^[0-9]{4}$`).MatchString(d.ID) {
		add("filename does not match NNNN-slug.md")
	}

	// 2. H1 title present.
	if strings.TrimSpace(d.Title) == "" {
		add("H1 title is missing or empty")
	}

	// 3-5. Load body for required-section + outcome check.
	body, err := s.decisionRepoBody(modelPath, d.ID) // see helper below
	if err != nil {
		add(fmt.Sprintf("failed to load body: %v", err))
		return issues
	}
	parsed, err := decisiondomain.ParseBody(body)
	if err != nil {
		add(fmt.Sprintf("failed to parse body: %v", err))
		return issues
	}

	if _, ok := parsed.Sections["context"]; !ok {
		add("missing required section: Context and Problem Statement")
	}
	if _, ok := parsed.Sections["options"]; !ok {
		add("missing required section: Considered Options")
	} else if len(parsed.Options) == 0 {
		add("Considered Options section has no bullets")
	}
	if _, ok := parsed.Sections["outcome"]; !ok {
		add("missing required section: Decision Outcome")
	}

	// 5. Decision Outcome shape (skipped when legacy-outcome is set).
	if d.Status == "accepted" && !d.LegacyOutcome {
		if parsed.ChosenOption == "" {
			add(`Decision Outcome must contain Chosen option: "..." when status is accepted`)
		} else if !slices.Contains(parsed.Options, parsed.ChosenOption) {
			add(fmt.Sprintf("chosen option %q is not in Considered Options", parsed.ChosenOption))
		}
	}

	// 6. Status vocabulary.
	if d.Status != "" && !statusRe.MatchString(d.Status) {
		add(fmt.Sprintf("status %q is not valid MADR vocabulary", d.Status))
	}

	// 7. Supersession forward integrity.
	if m := supersededRe.FindStringSubmatch(d.Status); m != nil {
		successorID := m[1]
		successor, ok := byID[successorID]
		if !ok {
			add(fmt.Sprintf("status references ADR-%s but no such ADR exists", successorID))
		} else if !slices.Contains(successor.Supersedes, d.ID) {
			add(fmt.Sprintf("ADR-%s status says it supersedes us, but its supersedes list does not include %s", successorID, d.ID))
		}
	}

	// 8. Supersession reverse integrity.
	for _, predID := range d.Supersedes {
		pred, ok := byID[predID]
		if !ok {
			add(fmt.Sprintf("supersedes %s but no such ADR exists", predID))
			continue
		}
		expected := fmt.Sprintf("superseded by ADR-%s", d.ID)
		if pred.Status != expected {
			add(fmt.Sprintf("supersedes %s but ADR-%s status is %q, not %q", predID, predID, pred.Status, expected))
		}
	}

	// 10. Comments integrity.
	for i, c := range d.Comments {
		if strings.TrimSpace(c.Text) == "" {
			add(fmt.Sprintf("comment %d has empty text", i+1))
		} else if _, err := strconv.Atoi(strings.TrimSpace(c.Text)); err == nil {
			add(fmt.Sprintf("comment %d has placeholder text %q; run adg migrate to recover", i+1, c.Text))
		}
	}

	return issues
}

}
```
